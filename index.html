async function fetchCalendarBusy() {

  if (!GCAL_URL || GCAL_URL === 'YOUR_APPS_SCRIPT_URL_HERE') return;

  try {

    const s = new Date(), e = new Date();

    e.setDate(s.getDate() + 50);

    const url = `${GCAL_URL}?action=busy&start=${s.toISOString()}&end=${e.toISOString()}`;

    const res = await fetch(url);

    const data = await res.json();

    if (data.busy) busyPeriods = data.busy.map(b => ({ s: new Date(b.start), e: new Date(b.end) }));

  } catch(ex) { console.log('Google Calendar 连接失败:', ex); }

}


function calConflict(date, t, durMin) {

  if (!busyPeriods.length) return false;

  const [h, m] = t.split(':').map(Number);

  const ss = new Date(date.getFullYear(), date.getMonth(), date.getDate(), h, m, 0);

  const se = new Date(ss.getTime() + durMin * 60000);

  return busyPeriods.some(b => ss < b.e && se > b.s);

}
